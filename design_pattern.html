public abstract class Entry {
    public string Name;
    public abstract int GetTotalSize();
    public abstract void FindDocx(List<string> results);
    public abstract string ToXml(int indent);
}

public class FileLeaf : Entry {
    public int Size;
    public override int GetTotalSize() => Size;
    public override void FindDocx(List<string> results) {
        if (Name.EndsWith(".docx")) results.Add(Name);
    }
    public override string ToXml(int indent) => 
        new string(' ', indent) + $"<{Name}></{Name}>";
}

public class DirComposite : Entry {
    public List<Entry> Children = new List<Entry>();
    public override int GetTotalSize() => Children.Sum(c => c.GetTotalSize());
    public override void FindDocx(List<string> res) {
        foreach(var child in Children) child.FindDocx(res);
    }
    public override string ToXml(int indent) {
        var sb = new StringBuilder();
        sb.AppendLine(new string(' ', indent) + $"<{Name}>");
        foreach(var child in Children) sb.AppendLine(child.ToXml(indent + 2));
        sb.AppendLine(new string(' ', indent) + $"</{Name}>");
        return sb.ToString();
    }
}
